public class Solution {
    public int FindDuplicate(int[] nums) {
  
        int size = nums.Length ;
        int [] flags= new int [size];
        
   
        for(int i =0;i<nums.Length;i++)
           flags[nums[i]]++;
        
          for(int i =0;i<nums.Length;i++)
          if(flags[i]>1)
        return i;
        
        
        return 0;
    }
}
