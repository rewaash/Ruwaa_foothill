public class Solution {
    public int[] TwoSum(int[] nums, int target) {
        
        int [] ans = new int [2] ;
        
        for(int i=0;i<nums.Length;i++){
              int x = target-nums[i];
            for(int j=0;j<nums.Length;j++){
          
                if(nums[j]==x)
                    if(i!=j){
                        
                    if(i>j) {
                        ans[0]=j;
                        ans[1] =i;}
                        else{ans[0]=i;
                        ans[1] =j;} 
                    }
       
            
            }}
        
        return ans;
    }
}
